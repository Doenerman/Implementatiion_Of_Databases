\begin{exercise}{4}

\begin{itemize}
  \item We need $1000$ pages to represent R
  \item We need $200$ pages to represent S
\end{itemize}

\begin{subexercise}
  \begin{displaymath}
  R + R \cdot S = 1000 + 1000\cdot 200 = 2001000
  \end{displaymath}
\end{subexercise}

\begin{subexercise}
  \begin{displaymath}
  R + \lceil R/(B-2)\rceil \cdot S = 1000 + \lceil\frac{1000}{50}\rceil\cdot 200= 5000
  \end{displaymath}
\end{subexercise}
\end{exercise}
